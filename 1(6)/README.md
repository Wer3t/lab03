2.3 Написання коду програми
```
int main (){
 char arr[n] = "viktoriia"; 
 cycle(arr); 
 return 0;
 }
 
 int cycle(char arr[n]){
int i=0, v=0 ;
while (arr[i] != '\0') {    // create a cycle for counting vowels
        switch (arr [i]){
        case 'a': v++; break;
        case 'e': v++; break;
        case 'i': v++; break;
        case 'o': v++; break;
        case 'u': v++; break;
        case 'y': v++; break;
 }
 i++;
 }
}
```
